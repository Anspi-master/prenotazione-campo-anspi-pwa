    // Funzione per ottimizzare le dimensioni dell'iframe
    function optimizeIframe() {
      const iframe = document.getElementById('appFrame');
      const container = document.getElementById('iframeContainer');
      
      if (!iframe || !container) return;
      
      const containerWidth = container.offsetWidth;
      const containerHeight = container.offsetHeight;
      
      // Assicurati che l'iframe riempia esattamente il container
      iframe.style.width = containerWidth + 'px';
      iframe.style.height = containerHeight + 'px';
      
      // Per dispositivi molto piccoli, potresti voler scalare il contenuto
      if (window.innerWidth < 360) {
        iframe.style.transform = 'scale(0.95)';
        iframe.style.transformOrigin = 'center center';
      } else {
        iframe.style.transform = 'scale(1)';
      }
    }
    
    // Gestione caricamento iframe
    document.getElementById('appFrame').addEventListener('load', function() {
      document.getElementById('iframeContainer').classList.add('iframe-loaded');
      optimizeIframe();
      
      // Prova a comunicare con l'iframe per ottimizzazioni aggiuntive
      try {
        this.contentWindow.postMessage({
          type: 'PWA_SHELL_READY',
          viewport: {
            width: window.innerWidth,
            height: window.innerHeight
          }
        }, '*');
      } catch (e) {
        console.log('Cross-origin communication not available');
      }
    });
    
    // Gestione ridimensionamento finestra
    let resizeTimeout;
    window.addEventListener('resize', function() {
      clearTimeout(resizeTimeout);
      resizeTimeout = setTimeout(() => {
        optimizeIframe();
        
        // Ricarica l'iframe se necessario dopo rotazione schermo
        const iframe = document.getElementById('appFrame');
        if (iframe && Math.abs(window.orientation) === 90) {
          // Landscape mode - potresti voler ricaricare
          setTimeout(() => {
            iframe.src = iframe.src;
          }, 300);
        }
      }, 250);
    });
    
    // Gestione orientamento schermo
    window.addEventListener('orientationchange', function() {
      setTimeout(() => {
        optimizeIframe();
        // Forza un refresh della viewport
        const viewport = document.querySelector('meta[name="viewport"]');
        if (viewport) {
          viewport.setAttribute('content', 
            'width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no, viewport-fit=cover'
          );
        }
      }, 500);
    });

    // Gestione installazione PWA (Chrome/Edge/Android)
    let deferredPrompt = null;
    const installBtn = document.getElementById('installBtn');
    
    window.addEventListener('beforeinstallprompt', (e) => {
      e.preventDefault();
      deferredPrompt = e;
      installBtn.style.display = 'inline-block';
    });
    
    installBtn?.addEventListener('click', async () => {
      if (!deferredPrompt) return;
      deferredPrompt.prompt();
      try { 
        const result = await deferredPrompt.userChoice; 
        console.log('Install prompt result:', result);
      } catch (err) {
        console.log('Install prompt error:', err);
      }
      installBtn.style.display = 'none';
      deferredPrompt = null;
    });

    // Rilevazione stato rete
    function updateOnlineStatus() {
      const app = document.getElementById('app');
      const isOffline = !navigator.onLine;
      app.classList.toggle('offline', isOffline);
      
      if (isOffline) {
        console.log('App is offline');
      } else {
        console.log('App is online');
      }
    }
    
    window.addEventListener('online', updateOnlineStatus);
    window.addEventListener('offline', updateOnlineStatus);
    updateOnlineStatus();

    // Previeni zoom accidentale
    document.addEventListener('touchstart', function(e) {
      if (e.touches.length > 1) {
        e.preventDefault();
      }
    }, { passive: false });
    
    let lastTouchEnd = 0;
    document.addEventListener('touchend', function(e) {
      const now = (new Date()).getTime();
      if (now - lastTouchEnd <= 300) {
        e.preventDefault();
      }
      lastTouchEnd = now;
    }, false);

    // Registra Service Worker
    if ('serviceWorker' in navigator) {
      window.addEventListener('load', function() {
        navigator.serviceWorker.register('./service-worker.js?ver=3')
          .then(function(reg) {
            console.log('âœ… SW registered', reg.scope);
            if (reg.waiting) { 
              reg.waiting.postMessage({ type: 'SKIP_WAITING' }); 
            }
            reg.addEventListener('updatefound', () => {
              const sw = reg.installing;
              if (!sw) return;
              sw.addEventListener('statechange', () => {
                if (sw.state === 'installed' && navigator.serviceWorker.controller) {
                  // Nuovo contenuto disponibile
                  if (confirm('Nuova versione disponibile. Ricaricare?')) {
                    location.reload();
                  }
                }
              });
            });
          })
          .catch(function(err) { 
            console.log('âŒ SW registration failed', err); 
          });
      });
    }
    
    // Inizializzazione
    document.addEventListener('DOMContentLoaded', function() {
      optimizeIframe();
      
      // Nascondi la barra degli indirizzi su mobile dopo un po'
      setTimeout(() => {
        window.scrollTo(0, 1);
      }, 1000);
    });
